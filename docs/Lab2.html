4.2 IQR approach¶

In [ ]:
Q1,Q3 = price.quantile([0.25,.75])
IQR = Q3 - Q1
lower_whisker = Q1 - 1.5 * IQR
upper_whisker = Q3 + 1.5 * IQR

lower_bound = price >= lower_whisker
upper_bound = price <= upper_whisker
iqr_data = data_AB[(lower_bound) & (upper_bound)]
